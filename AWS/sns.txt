# Set TOPIC
SNS_TOPIC="arn:aws:sns:eu-west-1:441687103692:SNS-Topic-NAME"

# List AS_GROUPS
as-describe-auto-scaling-groups |awk '/stack-name/ { print $2}'

# Set our new one
AS_GROUP="awseb-e-ku2trku6hc-stack-AWSEBAutoScalingGroup-J05RW3XLPQ0B"

# Configure notification
as-put-notification-configuration $AS_GROUP --topic-arn $SNS_TOPIC --notification-types autoscaling:EC2_INSTANCE_LAUNCH, autoscaling:EC2_INSTANCE_TERMINATE

# Describe
as-describe-notification-configurations

# O tambien con el siguiente comando de CLI3
aws autoscaling describe-notification-configurations --auto-scaling-group-name $AS_GROUP
